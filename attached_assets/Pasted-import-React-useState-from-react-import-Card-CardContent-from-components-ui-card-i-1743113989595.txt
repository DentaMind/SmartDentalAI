import React, { useState } from 'react';
import { Card, CardContent } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';
import { Label } from '@/components/ui/label';
import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group';

interface DiagnosisOption {
  id: string;
  label: string;
  reasoning: string;
  confidence: number;
}

interface DiagnosisFeedbackUIProps {
  options: DiagnosisOption[];
  onSubmit: (data: {
    selectedDiagnosisId: string;
    feedback: string;
  }) => void;
  onRequestMoreInfo?: () => void;
}

const DiagnosisFeedbackUI: React.FC<DiagnosisFeedbackUIProps> = ({ options, onSubmit, onRequestMoreInfo }) => {
  const [selectedId, setSelectedId] = useState<string>('');
  const [feedback, setFeedback] = useState<string>('');

  const handleSubmit = () => {
    if (selectedId) {
      onSubmit({ selectedDiagnosisId: selectedId, feedback });
    }
  };

  return (
    <Card className="p-4 shadow-xl border">
      <CardContent>
        <h2 className="text-xl font-semibold mb-4">Review AI Diagnosis</h2>

        <RadioGroup onValueChange={setSelectedId} value={selectedId} className="space-y-4">
          {options.map(option => (
            <div key={option.id} className="border rounded-lg p-3">
              <RadioGroupItem value={option.id} id={option.id} />
              <Label htmlFor={option.id} className="ml-2 font-medium">
                {option.label} ({Math.round(option.confidence * 100)}% confidence)
              </Label>
              <p className="text-sm text-muted-foreground mt-1 italic">Reasoning: {option.reasoning}</p>
            </div>
          ))}
        </RadioGroup>

        <div className="mt-6">
          <Label htmlFor="feedback">Feedback (optional)</Label>
          <Textarea
            id="feedback"
            placeholder="Explain why this diagnosis is correct or incorrect."
            className="mt-2"
            value={feedback}
            onChange={(e) => setFeedback(e.target.value)}
          />
        </div>

        <div className="flex gap-4 mt-6">
          <Button onClick={handleSubmit} disabled={!selectedId}>
            Submit Feedback
          </Button>
          <Button variant="outline" onClick={onRequestMoreInfo}>
            Request More Info
          </Button>
        </div>
      </CardContent>
    </Card>
  );
};

export default DiagnosisFeedbackUI;
