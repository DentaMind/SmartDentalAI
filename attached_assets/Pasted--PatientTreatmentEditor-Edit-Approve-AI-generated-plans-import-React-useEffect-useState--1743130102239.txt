// PatientTreatmentEditor: Edit + Approve AI-generated plans

import React, { useEffect, useState } from "react";
import axios from "axios";
import { Card } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";
import { Input } from "@/components/ui/input";
import { Badge } from "@/components/ui/badge";
import { useToast } from "@/components/ui/use-toast";

export default function PatientTreatmentEditor({ patientId }: { patientId: string }) {
  const [treatmentPlan, setTreatmentPlan] = useState<any>(null);
  const [editedPlan, setEditedPlan] = useState<string>("");
  const [loading, setLoading] = useState(false);
  const { toast } = useToast();

  useEffect(() => {
    axios.get(`/api/treatment-plan/${patientId}`).then((res) => {
      setTreatmentPlan(res.data);
      setEditedPlan(res.data?.planText || "");
    });
  }, [patientId]);

  const savePlan = async () => {
    setLoading(true);
    try {
      await axios.post(`/api/treatment-plan/${patientId}/approve`, {
        finalPlan: editedPlan
      });
      toast({ title: "Treatment Plan Saved" });
    } catch (err) {
      toast({ title: "Error Saving", description: "Try again later." });
    } finally {
      setLoading(false);
    }
  };

  if (!treatmentPlan) return <div>Loading treatment plan...</div>;

  return (
    <Card className="p-4 space-y-4">
      <h2 className="text-xl font-semibold">Treatment Plan</h2>
      <div className="text-sm text-muted-foreground mb-2">
        <strong>AI Confidence:</strong> {treatmentPlan.confidence}%
      </div>
      <div className="text-sm text-muted-foreground mb-4">
        <strong>Reasoning:</strong> {treatmentPlan.reasoning}
      </div>

      <Textarea
        value={editedPlan}
        onChange={(e) => setEditedPlan(e.target.value)}
        className="min-h-[160px]"
      />

      <Button onClick={savePlan} disabled={loading}>
        {loading ? "Saving..." : "Save & Approve Plan"}
      </Button>
    </Card>
  );
}
