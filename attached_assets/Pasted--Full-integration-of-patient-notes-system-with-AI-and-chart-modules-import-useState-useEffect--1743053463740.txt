// Full integration of patient notes system with AI and chart modules

import { useState, useEffect } from 'react';
import { getPatientChartData, submitNoteForApproval } from '@/lib/api';
import { useVoiceToText } from '@/lib/voice-ai';
import { usePatient } from '@/hooks/use-patient';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';
import { TemplatesDropdown } from '@/components/patient/templates-dropdown';

export default function PatientNotesPanel({ patientId }) {
  const { patient } = usePatient(patientId);
  const [notes, setNotes] = useState('');
  const [chartData, setChartData] = useState(null);
  const [template, setTemplate] = useState('');
  const { transcript, startListening, stopListening, reset } = useVoiceToText();

  useEffect(() => {
    getPatientChartData(patientId).then(setChartData);
  }, [patientId]);

  useEffect(() => {
    if (transcript) setNotes((prev) => prev + ' ' + transcript);
  }, [transcript]);

  const handleTemplateInsert = (templateText) => {
    setNotes((prev) => prev + '\n\n' + templateText);
  };

  const handleSubmit = async () => {
    await submitNoteForApproval({
      patientId,
      providerId: patient?.providerId,
      notes,
      date: new Date().toISOString(),
    });
    alert('Note submitted for approval. It will appear after review.');
    setNotes('');
  };

  return (
    <div className="p-4 space-y-4">
      <h2 className="text-xl font-bold">Patient Notes</h2>
      <TemplatesDropdown onSelect={handleTemplateInsert} />

      <Textarea
        className="w-full h-64"
        value={notes}
        onChange={(e) => setNotes(e.target.value)}
        placeholder="Type or use voice input to create a note..."
      />

      <div className="flex gap-4">
        <Button onClick={startListening}>üéôÔ∏è Start Voice</Button>
        <Button onClick={stopListening}>‚èπ Stop</Button>
        <Button variant="secondary" onClick={reset}>Reset</Button>
        <Button onClick={handleSubmit} className="ml-auto">Submit for Approval</Button>
      </div>
    </div>
  );
}
