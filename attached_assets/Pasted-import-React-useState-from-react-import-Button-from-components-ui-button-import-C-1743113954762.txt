import React, { useState } from "react";
import { Button } from "@/components/ui/button";
import { Card, CardHeader, CardContent } from "@/components/ui/card";
import { Textarea } from "@/components/ui/textarea";
import { Label } from "@/components/ui/label";
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group";

interface DiagnosisOption {
  id: string;
  label: string;
  confidence: number;
  reasoning: string;
}

interface DiagnosisFeedbackUIProps {
  patientId: string;
  diagnosisOptions: DiagnosisOption[];
  onSubmit: (feedback: DiagnosisFeedbackSubmission) => void;
}

export interface DiagnosisFeedbackSubmission {
  selectedDiagnosisId: string;
  feedback: string;
  needsMoreInfo: boolean;
}

const DiagnosisFeedbackUI: React.FC<DiagnosisFeedbackUIProps> = ({
  patientId,
  diagnosisOptions,
  onSubmit,
}) => {
  const [selectedDiagnosisId, setSelectedDiagnosisId] = useState<string>("");
  const [feedback, setFeedback] = useState<string>("");
  const [needsMoreInfo, setNeedsMoreInfo] = useState<boolean>(false);

  const handleSubmit = () => {
    if (!selectedDiagnosisId && !needsMoreInfo) return;
    onSubmit({ selectedDiagnosisId, feedback, needsMoreInfo });
  };

  return (
    <Card className="mt-6">
      <CardHeader>
        <h3 className="text-xl font-semibold">Review AI Diagnosis</h3>
      </CardHeader>
      <CardContent>
        <RadioGroup
          onValueChange={setSelectedDiagnosisId}
          value={selectedDiagnosisId}
          className="space-y-4"
        >
          {diagnosisOptions.map((option) => (
            <div
              key={option.id}
              className="border rounded-md p-4 hover:bg-muted transition"
            >
              <RadioGroupItem value={option.id} id={option.id} />
              <Label htmlFor={option.id} className="ml-2 font-medium">
                {option.label} (Confidence: {option.confidence}%)
              </Label>
              <p className="text-sm text-muted-foreground mt-1">
                {option.reasoning}
              </p>
            </div>
          ))}
        </RadioGroup>

        <div className="mt-4">
          <Label htmlFor="feedback">Doctor Feedback</Label>
          <Textarea
            id="feedback"
            placeholder="Explain why this diagnosis is correct/incorrect or request clarification."
            value={feedback}
            onChange={(e) => setFeedback(e.target.value)}
          />
        </div>

        <div className="mt-4">
          <label className="flex items-center space-x-2">
            <input
              type="checkbox"
              checked={needsMoreInfo}
              onChange={() => setNeedsMoreInfo((prev) => !prev)}
            />
            <span className="text-sm">AI needs more information</span>
          </label>
        </div>

        <Button onClick={handleSubmit} className="mt-4">
          Submit Feedback
        </Button>
      </CardContent>
    </Card>
  );
};

export default DiagnosisFeedbackUI;
